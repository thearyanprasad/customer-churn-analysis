import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import os

os.makedirs(r"C:\Users\user\Desktop\customer-churn-analysis\visuals", exist_ok=True)

df = pd.read_csv(
    r"C:\Users\user\Desktop\customer-churn-analysis\data\churn_data.csv"
)

df['TotalCharges'] = pd.to_numeric(df['TotalCharges'], errors='coerce')
df.dropna(subset=['TotalCharges'], inplace=True)

df['Churn'] = df['Churn'].map({'Yes': 1, 'No': 0})

df.drop('customerID', axis=1, inplace=True)

print("=" * 50)
print("Dataset Shape:", df.shape)
print("=" * 50)

print("\nColumn Names:")
print(df.columns)

print("\nMissing Values:")
print(df.isnull().sum())

print("\nChurn Distribution:")
print(df['Churn'].value_counts())

churn_rate = df['Churn'].mean() * 100
print(f"\nOverall Churn Rate: {churn_rate:.2f}%")

contract_churn = df.groupby('Contract')['Churn'].mean() * 100

plt.figure(figsize=(8, 5))
contract_churn.plot(
    kind='bar',
    color='#FF6B35',
    edgecolor='black'
)

plt.title('Churn Rate by Contract Type')
plt.ylabel('Churn Rate (%)')
plt.xlabel('')
plt.xticks(rotation=0)
plt.tight_layout()

plt.savefig(
    r"C:\Users\user\Desktop\customer-churn-analysis\visuals\churn_by_contract.png"
)
plt.show()

plt.figure(figsize=(10, 5))

sns.histplot(
    data=df,
    x='tenure',
    hue='Churn',
    bins=30,
    multiple='stack',
    palette={0: '#111111', 1: '#FF6B35'})

plt.title('Customer Tenure vs Churn')
plt.xlabel('Tenure (Months)')
plt.ylabel('Customers')

plt.tight_layout()

plt.savefig(
    r"C:\Users\user\Desktop\customer-churn-analysis\visuals\churn_by_tenure.png"
)
plt.show()

plt.figure(figsize=(8, 5))

sns.boxplot(
    data=df,
    x='Churn',
    y='MonthlyCharges',
    hue='Churn',
    palette={0: '#111111', 1: '#FF6B35'},
    legend=False

plt.title('Monthly Charges: Churned vs Retained')
plt.xlabel('')
plt.ylabel('Monthly Charges')

plt.xticks(
    [0, 1],
    ['Stayed', 'Churned']

plt.tight_layout()

plt.savefig(
    r"C:\Users\user\Desktop\customer-churn-analysis\visuals\charges_vs_churn.png"
